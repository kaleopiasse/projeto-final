#Inicializando projeto

npm init -y

npm i restify --save -E
npm i @types/restify -D -E

npm i nodemon -g
nodemon dist/main.js

npm i mongoose -P -E
npm i @types/mongoose -D -E

#Errors
npm i restify-errors -P -E
npm i @types/restify-errors -D -E

#Encriptar password
npm i bcrypt -P -E
npm i @types/bcrypt -D -E

#Autenticação
npm i jsonwebtoken -P -E
npm i @types/jsonwebtoken -D -E

#Cors
npm i restify-cors-middleware -P -E
npm i @types/restify-cors-middleware -D -E

#Criando dump da database
/usr/bin$ mongodump --db pdi-api --out /home/kaleo/Projetos/projeto-final/backend/dump

#Fazendo restore da database, antes de executar é necessario criar a database
/usr/bin$ mongorestore --db test-dump /home/kaleo/Projetos/projeto-final/backend/dump/pdi-api
